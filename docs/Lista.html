def dieta_gases_anuales(gramos)
        aux = @head
        suma_gases = 0
        i = 0
        while (!aux.nil?)
                suma_gases += aux[:value].auxiliar(gramos[i])
                aux = aux[:next]
                i = i+1
        end
        return suma_gases.round(2)

end

def extract_head()
        if @head.nil?
                raise RuntimeError, "La lista está vacía"
        else
                if @head == @tail
                        @head, @tail = nil
                else
                        @head = @head[:next]
                        @head[:prev] = nil
                end

        end
end

def insert_head(value)
        if @head.nil?
                @head = Nodo.new(value, nil, nil)
                @tail = @head
        else
                @head[:prev] = Nodo.new(value, @head, nil)
                @head = @head[:prev]
        end
end

def insert_tail(value)
        if @head.nil?
                @head = Nodo.new(value, nil, nil)
                @tail = @head
        else
                @tail[:next] = Nodo.new(value, nil, @tail)
                @tail = @tail[:next]
        end
end
