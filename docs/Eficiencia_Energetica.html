def emisiones_gases_anuales
        aux = @lista_alimentos.head
        aux2 = @lista_gramos.head
        suma_gases_anuales = 0
        while (!aux.nil?)
                suma_gases_anuales += aux[:value].auxiliar(aux2[:value])
                aux = aux[:next]
                aux2 = aux2[:next]
        end

        return suma_gases_anuales.round(2)
end
